def vector(article,vocab,indices,N):
    v = [0 for i in range(len(indices))]
    for word in article.vocabulary().keys():
        tf = article.vocabulary()[word]
        idf = np.log10(N/vocab[word] + 0.01 )
        w = tf*idf
        v[indices[word]] = w 
    v = np.array(v)

def similarity(v1,v2):
    assert len(v1) == len(v2)
    return 1000*min(np.dot(v1,v2)/(np.linalg.norm(v1)*np.linalg.norm(v2)),1.0)

class Event:
    def vocabulary(self, dummy_force=False):
        return self._vocabulary
        
    def __init__(self):
        self.article_count = 0
        self.articles = []
        self.docvector = []
        self._vocabulary = {}
    
    # add updates the docvector automaticlally
    def add(self, article, similarity):
        assert(isinstance(article, NewsArticle))
        article.similarity = similarity
        self.articles.append(article)
        self.article_count+=1
        article_vocab = article.vocabulary()
        for i in article_vocab:
            if i in self._vocabulary:
                self._vocabulary[i]+=article_vocab[i]
            else:
                self._vocabulary[i]=article_vocab[i]
        self._article_id_count = Counter([article._event_id for article in self.articles])
        self._event_id = max(self._article_id_count, key=lambda x: self._article_id_count[x])
    
    def update_vector(self, vocab, indices, N):
        self.docvector = vector(self,vocab,indices,N)[:]

    def __str__(self):
        return "Event %s, Total articles %s\n" %(self._event_id, self.article_count) +\
                ('\n'.join(str((article.title[:30]+'..', article._event_id[:10]+'..', article.similarity)) for article in sorted(self.articles,key=lambda x:x.similarity,reverse=True)) if self.article_count!=0 else "No article found") + \
                '\n' + str(self._article_id_count)
    
    def __repr__(self):
        return self.__str__()

def metrics(articles, events, CE=CE, CF=CF):
    # articles related to a topic
    actual_event_articles = {}
    for article in articles:
        if article._event_id in actual_event_articles:
            actual_event_articles[article._event_id].append(article)
        else:
            actual_event_articles[article._event_id]=[article]
    overall_cost = 0
    for event in events:    
        A = len(actual_event_articles[event._event_id])
        A1 = event._article_id_count[event._event_id]
        B = len(articles) - A
        B1 = event.article_count - A1
        PE = B1/B
        PF = (A-A1)/A
        #print("PE, Pf = ", PE,PF)
        P= A/len(articles)
        overall_cost += (CE*PE*P + CF*PF*(1-P))**2
    return math.sqrt(overall_cost/len(events))

def cluster_articles(articles, alpha_l, beta_l, threshold):
    vocab, indices = overall_vocab(articles, True)
    global alpha
    global beta
    alpha = alpha_l
    beta = beta_l
    for article in articles :
        article.docvector = vector(article, vocab=vocab, indices=indices, N=len(articles))
    events = []
    for article in articles:
        # event, similarity, e_index
        #print(events,end='\n\n\n')
        potential = (None,-1,-1)
        for i,event in enumerate(events):
            cur_sim = similarity(event.docvector, article.docvector)
            if potential[1]<cur_sim: 
                potential = (event,cur_sim,i)
        if(potential[1]>=threshold):
            #print("adding article: %s to event"% article.title,potential[2])
            potential[0].add(article,potential[1])
            potential[0].update_vector(vocab, indices, len(articles))
        else:
            #print("Creating New Event for article:", article.title)
            e = Event()
            e.add(article,similarity(article.docvector, article.docvector))
            e.update_vector(vocab, indices, len(articles))
            events.append(e)
        article._event_assigned=potential[2]
    return events
